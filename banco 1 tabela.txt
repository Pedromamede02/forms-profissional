CREATE DATABASE projeto;

USE projeto;

CREATE TABLE tbusu (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    sobrenome VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL
);


Parte 1 


CREATE TABLE Avaliacoes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT,
    avaliacao1 INT CHECK (avaliacao1 BETWEEN 0 AND 10),
    avaliacao2 INT CHECK (avaliacao2 BETWEEN 0 AND 10),
    avaliacao3 INT CHECK (avaliacao3 BETWEEN 0 AND 10),
    avaliacao4 INT CHECK (avaliacao4 BETWEEN 0 AND 10),
    avaliacao5 INT CHECK (avaliacao5 BETWEEN 0 AND 10),
    avaliacao6 INT CHECK (avaliacao6 BETWEEN 0 AND 10),
    avaliacao7 INT CHECK (avaliacao7 BETWEEN 0 AND 10),
    avaliacao8 INT CHECK (avaliacao8 BETWEEN 0 AND 10),
    data_submissao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES tbusu(id)
);
